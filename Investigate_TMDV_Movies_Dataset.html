def HighLow(x):
    Low_index = df_movie[x].idxmin()
    High_index = df_movie[x].idxmax()
    lowest = pd.DataFrame(df_movie.loc[Low_index,:])
    highest = pd.DataFrame(df_movie.loc[High_index,:])
    
    #Print the highest and lowest
    print("Highest " +  x +  " : ", df_movie["original_title"][High_index])
    print("Lowset " + x + " : ", df_movie["original_title"][Low_index])
    return pd.concat([highest,lowest], axis=1)
